//4的幂
//给定一个32位的整数, 判断其是否是4的幂次方.

//4 = 2^2 ---> 和2的幂次相似 
//1 = 1 | 4 = 100 | 16 = 10000 ..... 发现1总是出现在奇数位上

	bool isPowerOfFour(int num) { // 注意是32位有符号数
        
  		// 4 & 3 100 & 011
  		// 8 & 7 1000 & 0111 
        if(num <= 0 || num & num - 1) return false;
        
        //我们发现4的幂 也一定可以转换成2的幂 2 ---> 转换为0/1进制, 并且1一定出现在奇数位
        //4 100  --- 用来检测奇数位的 0x5 = 0101 
        
        //1出现在奇数位 则0x55555555仍为原数
        return num & 0x55555555; // num % 3 == 1
    }
