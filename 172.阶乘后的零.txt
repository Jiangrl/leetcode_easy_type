//阶乘后的零
//给定一个整数n, 返回n!结果尾数中零的数量.
	
	int trailingZeroes(int n) {
        
        //算最终的阶乘结果可能会导致过界
        //怎么样乘 结果的尾部有0: 0 / 2 / 4 / 8 * 5 

        //2*5 结尾为0 4 = 2 * 2 ... 依次类推 10 = 2 * 5
        //按照这种拆分的方式 我们知道2.number >= 5.number
        //即有5必然有2 所以计算5的个数即可!

        int count = 0;
        while(n > 4){
            count += n / 5;
            n /= 5;
        }
        return count;
    }
