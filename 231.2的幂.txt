//2的幂
//给定一个整数, 判断它是否是2的幂次方

	//2的幂有一个特点，二进制形式只有一个1，所以题目转化为判断其二进制是否只有一个1。
	//2 10 | 4 100 | 8 1000 | 16 10000....
	//2 1 10 & 01 = 0 | 4 3 100 & 011 = 0...
    //去掉二进制最后一个1的方法是n&(n-1)。

    //位运算中的一些方法是很简单高效的!
    bool isPowerOfTwo(int n) {
        
        if(n <= 0) return false;
        if((n & n - 1) == 0) return true;
        return false;
    }
