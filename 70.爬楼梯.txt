//爬楼梯
//每次只能走1或者2步 然后n阶到达楼顶
//求解到达楼顶的上楼梯方法种类是多少

	/*
    这题可以直接递归，但是这样复杂度就大了， 可以使用一个一维数组存储， 可以爬1 / 2步
    */
    int climbStairs(int n) {
        int a[n + 1];
        
        a[0] = 1, a[1] = 1;
        for(int i = 2; i <= n; i ++){
            
            a[i] = a[i - 1] + a[i - 2];
        }
        
        return a[n];
    }
